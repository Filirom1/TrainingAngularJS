Controllers inheritance

Controller inheritance is base on $scope inheritance.

angular.module('sample', []) .controller('ParentCtrl', function($scope) { $scope.firstName = 'Bart'; }) .controller('ChildCtrl', function ($scope) { $scope.firstName = 'Bob'; $scope.lastName = 'Sponge'; }) .controller('BabyCtrl', function ($scope) { $scope.lastName="Marley"; }); <div ng-app="sample" > <div style="border: solid 2px" ng-controller="ParentCtrl"> <p>Hello from Parent, {{firstName + " " + lastName}}</p> <div style="border: solid red 2px" ng-controller="ChildCtrl"> <p>Hello from Child, {{firstName + " " + lastName}}</p> <div style="border: solid yellow 2px" ng-controller="BabyCtrl"> <p>Hello from Baby, {{firstName + " " + lastName}}</p> </div> </div> </div> </div>
