Select

  • Angular select directive allows data-binding
  • Attribute ngOptions can be used to dynamically to generate a list of <option> elements.
<div ng-controller="SelectCtrl"> <select ng-model="user" ng-options="user.name group by user.country for user in users"> <option value="">-- Choose user --</option> </select> Selected: {{user}} </div> angular.module('trainingAJ') .exoController('SelectCtrl', ['$scope', function($scope) { $scope.users = [ { id: 1, name:'Bob', country:'France'}, { id: 2, name:'Bill', country:'England'}, { id: 3, name:'Luc', country:'France'}, { id: 4, name:'Joe', country:'France'}, { id: 5, name:'Matt', country:'Spain'} ]; }]);
