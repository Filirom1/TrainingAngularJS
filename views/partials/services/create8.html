value() and constant()

  • They are very similar
  • They don't have access to other angular services (no DI)
  • The main difference is that constant can be used as a provider during the configuration phase
angular.module('sample', []); // Simple constant .value('nbProductsPerPage', 5); // Utility methods library, but DI is not available .value('ProductUtils', function() { //TODO }); angular.module('sample', []) .constant('helloMessage', 'Hello world') //constant service is available during the config //Note that it is not suffixed by provider... .config(['myServiceProvider','helloMessage',function(myServiceProvider, helloMessage) { myServiceProvider.setMessage(helloMessage); }]);
